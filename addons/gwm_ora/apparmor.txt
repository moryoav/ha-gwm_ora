#include <tunables/global>

profile gwm_ora flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Network access for the internal API listener, Supervisor discovery, and GWM cloud calls.
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,

  # Graceful container shutdown.
  signal (send, receive) set=(kill,term,int,hup,cont),

  # Allow runtime reads within the container, while keeping writes scoped below.
  /** r,
  deny /proc/kcore rwklx,
  deny /proc/sysrq-trigger rwklx,

  # .NET runtime and application.
  /usr/bin/dotnet rix,
  /usr/share/dotnet/dotnet rix,
  /usr/share/dotnet/** mr,
  /app/ r,
  /app/** mr,
  /lib/** mr,
  /usr/lib/** mr,
  /bin/** rix,
  /usr/bin/** rix,

  # Runtime configuration, certificates, DNS, and container identity files.
  /etc/hosts r,
  /etc/hostname r,
  /etc/resolv.conf r,
  /etc/nsswitch.conf r,
  /etc/passwd r,
  /etc/group r,
  /etc/ssl/** r,

  # Generated add-on state, tokens, options.json, and temporary files.
  /data/ rw,
  /data/** rwk,
  /tmp/ rw,
  /tmp/** rwk,
  /run/ rw,
  /run/** rwk,

  # Basic devices used by the runtime.
  /dev/null rw,
  /dev/zero r,
  /dev/random r,
  /dev/urandom r,
}
